Supabase Schema
sql-- Student profile
students (
  id uuid PK,
  email text,
  degree text,          -- BS/MS/PhD
  program text,         -- CS, EE etc
  semester int,
  cgpa float,
  skills text[],        -- ['Python', 'ML', 'React']
  interests text[],
  preferred_types text[], -- ['Scholarship', 'Internship']
  financial_need bool,
  location_pref text,   -- Local/National/International
  nationality text,
  graduation_date date,
  gmail_refresh_token text,
  created_at timestamp
)

-- Raw emails pulled from Gmail or pasted
raw_emails (
  id uuid PK,
  student_id uuid FK,
  gmail_message_id text,  -- null if manually pasted
  subject text,
  sender text,
  sender_domain text,
  received_at timestamp,
  raw_body text,
  cleaned_body text,
  ingested_at timestamp
)

-- After classification
opportunities (
  id uuid PK,
  raw_email_id uuid FK,
  student_id uuid FK,
  is_opportunity bool,
  confidence float,       -- classifier confidence 0-1
  status text,            -- 'active' | 'expired' | 'ineligible'
  
  -- Extracted fields
  opp_type text,          -- Scholarship/Internship/Fellowship etc
  org_name text,
  deadline date,
  deadline_ambiguous bool,
  eligibility_raw text,   -- raw extracted text
  cgpa_required float,    -- null if not stated
  degree_required text,
  skills_required text[],
  documents_required text[],
  benefits text,
  funding_type text,      -- Full/Partial/None
  geo_scope text,         -- Local/National/International
  application_link text,
  
  -- Scores
  profile_fit_score float,
  urgency_score float,
  value_score float,
  semantic_score float,
  final_score float,
  
  -- Output
  explanation text,
  action_checklist text[],
  urgency_flag text,      -- Red/Orange/Yellow/Green
  
  created_at timestamp
)

-- Org knowledge base (pre-seeded, static)
org_knowledge (
  id uuid PK,
  org_name text,
  description text,       -- what gets embedded
  prestige_score int,     -- 1-10, you set this manually
  known_funding_type text,
  typical_scope text
)

-- Gmail sync tracking
gmail_sync_log (
  id uuid PK,
  student_id uuid FK,
  synced_at timestamp,
  emails_fetched int,
  opportunities_found int,
  last_message_id text    -- for pagination/dedup
)



Supabase pgvector tables
-- Requires: create extension if not exists vector;
-- All embeddings are 1536-dim (openai/text-embedding-3-small via GitHub Models).
-- Each table has an HNSW index with vector_cosine_ops for similarity search.

opportunity_chunks (
  id uuid PK,
  opportunity_id uuid FK -> opportunities(id) on delete cascade,
  student_id uuid FK -> students(id) on delete cascade,
  chunk_type text check (chunk_type in ('header','eligibility','benefits','documents','deadline')),
  chunk_text text,
  embedding vector(1536),
  deadline_date date,
  opp_type text,
  final_score float,
  created_at timestamp default now()
)

student_profile_embeddings (
  student_id uuid PK FK -> students(id) on delete cascade,
  narrated_text text,
  embedding vector(1536),
  updated_at timestamp default now()
)

org_knowledge_embeddings (
  org_knowledge_id uuid PK FK -> org_knowledge(id) on delete cascade,
  embedding vector(1536),
  updated_at timestamp default now()
)

RPCs (SQL functions for similarity search)
- match_opportunity_chunks(query_embedding vector, match_student_id uuid, match_count int)
    returns top-N chunks for a given student_id, ranked by 1 - (embedding <=> query_embedding)
- match_org_knowledge(query_embedding vector, match_count int)
    returns top-N org rows joined with org_knowledge, ranked by cosine similarity
